## DragModal

基于 Modal 的二次封装，可拖拽的浮层

<Playground>
  {class Example extends React.Component {
    constructor(props) {
      this.state = {
        visible: false,
        visible2: false,
      };
    }

    show() {
      this.setState({ visible: true });
    }

    onCancel() {
      this.setState({ visible: false });
    }

    show2() {
      this.setState({ visible2: true });
    }

    onCancel2() {
      this.setState({ visible2: false });
    }

    render () {
      const { visible, visible2 } = this.state;

      return (
        <React.Fragment>
          <LinkBtn onClick={() => this.show()}>Show-1</LinkBtn>
          <br/>
          <LinkBtn onClick={() => this.show2()}>Show-2</LinkBtn>
          <DragModal title="Modal-1" visible={visible} onCancel={() => this.onCancel()} />
          <DragModal title="Modal-2" visible={visible2} onCancel={() => this.onCancel2()} />
        </React.Fragment>
      )
    }
  }}
</Playground>
